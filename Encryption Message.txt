  cipherText = ""

  for ch in message:

    if ch.isalpha():


      if operation:

        stayInAlphabet = ord(ch) + shift

      else:

        stayInAlphabet = ord(ch) - shift


      if stayInAlphabet > ord('z'):

        stayInAlphabet -= 26

      finalLetter = chr(stayInAlphabet)

      cipherText += finalLetter

  if operation:

    print ("Your ciphered text is: ", cipherText)

  return cipherText
